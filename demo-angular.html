import { GlassMorphismService } from 'glassmorphic-kit';

@Component({
  selector: 'app-root',
  template: `
    <glass-morphism [variant]="selectedVariant">
      <h1>{{ title }}</h1>
      <glass-input 
        [(ngModel)]="inputValue"
        placeholder="Enter text...">
      </glass-input>
      <glass-button (click)="onSubmit()">
        Submit
      </glass-button>
    </glass-morphism>
  `
})
export class AppComponent {
  selectedVariant = 'medium';
  inputValue = '';

  constructor(private glassService: GlassMorphismService) {}
}
